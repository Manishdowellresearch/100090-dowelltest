        },

        },
        "second_user_selection":{
            "eventId":"call event fuction",
            "second_variable":"the second variable that user select from the selected_variable_for_pemutation stage",

            "Result":"result"
        },
        "Result":{
            "eventId":"call event fuction",

        }
    }
}
